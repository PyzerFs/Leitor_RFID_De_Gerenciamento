        IF WS-SALDO >= WS-TARIFA
            SUBTRACT WS-TARIFA FROM WS-SALDO

        ELSE
            DISPLAY "[COBOL] ❌ Saldo insuficiente!"
        END-IF
    ELSE
        DISPLAY "[COBOL] ❌ Cartão inválido!"
    END-IF.
